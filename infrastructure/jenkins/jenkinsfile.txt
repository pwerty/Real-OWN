pipeline {
    agent {
        kubernetes {
            yaml """
apiVersion: v1
kind: Pod
spec:
  containers:
  - name: unity
    image: 'unityci/editor:ubuntu-6000.0.24f1-windows-mono-3.1.0'
    command:
    - cat
    tty: true
    volumeMounts:
    - name: workspace-volume
      mountPath: /home/jenkins/agent # 작업 공간 고정
  volumes:
  - name: workspace-volume
    persistentVolumeClaim:
      claimName: jenkins
"""
        }
    }
    
    environment {
        HOME = "${WORKSPACE}"
        PROJECT_PATH = 'unity/elysianforest'
    }

    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/pwerty/Real-OWN.git', branch: 'main'
            }
        }

        stage('Unity Build') {
            steps {
                container('unity') {
                    withCredentials([string(credentialsId: 'UNITY_LICENSE', variable: 'LICENSE_DATA')]) {
                        // 분리한 스크립트 실행
                        sh "chmod +x ./infrastructure/jenkins/tools/unity-build.sh"
                        sh "./infrastructure/jenkins/tools/unity-build.sh"
                    }
                }
            }
            post {
                success {
                    archiveArtifacts artifacts: "${PROJECT_PATH}/Builds/Linux/**", fingerprint: true
                }
            }
        }
    }
}
